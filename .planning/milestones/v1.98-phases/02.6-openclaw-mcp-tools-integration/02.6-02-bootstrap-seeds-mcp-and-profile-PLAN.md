    ```python
    def _run_preseed(tmp_path: Path, env_overrides: dict[str, str]) -> Path:
        """Extract the pre-seed python heredoc from scripts/openclaw-bootstrap.sh
        and exec it with a merged env against a tmp config dir.

        Returns the path to the generated openclaw.json.
        """
        import os, re, subprocess, textwrap
        script_path = Path(__file__).parent.parent / "scripts" / "openclaw-bootstrap.sh"
        script_src = script_path.read_text()

        # Slice the pre-seed heredoc body (between `python3 - <<'PY'` and the matching `PY`).
        # The executor re-checks fence positions against the current file — the regex is
        # tolerant of surrounding shell quoting since the sh -lc wrapper uses '"'"'PY'"'"'.
        m = re.search(r"python3 - <<'?\"?'?PY'?\"?'?\s*\n(.+?)\n\s*PY(?:\b|'|\")", script_src, re.S)
        assert m, "could not locate pre-seed python heredoc in openclaw-bootstrap.sh"
        heredoc_body = m.group(1)

        # Redirect the heredoc's output path: replace the hard-coded
        # /home/node/.openclaw/openclaw.json with tmp_path / openclaw.json.
        target_json = tmp_path / "openclaw.json"
        patched = heredoc_body.replace(
            "/home/node/.openclaw/openclaw.json", str(target_json)
        )
        # Also make any shutil.copytree / os.makedirs calls land under tmp_path.
        patched = patched.replace("/home/node/.openclaw", str(tmp_path))

        script_file = tmp_path / "preseed.py"
        script_file.write_text(patched)

        # Build env. Require the same vars the heredoc reads, with sane defaults so
        # individual tests only need to override what they care about.
        env = {
            "PROVIDER_ID": "kimi",
            "PROVIDER_BASE_URL": "http://example.test",
            "AGENT_IDS_CSV": "agent-0",

            "ROBOCLAWS_MCP_URL": "http://host.docker.internal:18788/mcp",
            "ROBOCLAWS_TOOL_PROFILE": "minimal",

        env.update(env_overrides)
        # Inherit minimal system env for python3 availability
        for k in ("PATH", "HOME", "LANG"):
            if k in os.environ:
                env[k] = os.environ[k]

        subprocess.run(
            ["python3", str(script_file)],
            env=env, check=True, cwd=str(tmp_path),
            capture_output=True, text=True,
        )
        return target_json
    ```
